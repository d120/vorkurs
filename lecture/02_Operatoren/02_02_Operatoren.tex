				& 		& 50 >= 50 		& True \\ \hline
				kleiner als 			& < 	& 100 < 50; 	& False \\
				& 		& 50 < 50; 		& False \\ \hline
				kleiner als oder gleich & <= 	& 100 <= 50; 	& False \\
				&  		& 50 <= 50; 	& True \\ \hline
